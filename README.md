# 将数字转 千分位分隔

# 安装

```sh
npm i number-split
```

# 用法

```ts
import { numberSplit } from "number-split";

numberSplit(1234567.123456789, 3); // "1,234,567.123,456,789"
numberSplit("-1234567890.123456789", 3); // "-1,234,567,890.123,456,789"

numberSplit(1234567.123456789, 4); // "123,4567.1234,5678,9"
numberSplit("-1234567890.123456789", 4); // "-12,3456,7890.1234,5678,9"
```
